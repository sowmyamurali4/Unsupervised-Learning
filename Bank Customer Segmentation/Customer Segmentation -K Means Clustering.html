clusters = range(1, 11)  # a range of possible K values
meandistortions = []  # empty list for elbow curve plot

for k in clusters:
    model = KMeans(n_clusters=k)
    model.fit(df_scaled_subset)
    model.predict(df_scaled_subset)
    distortion = (
        sum(
            np.min(cdist(df_scaled_subset, model.cluster_centers_, "euclidean"), axis=1)
        )
        / df_scaled_subset.shape[0]
    )
    meandistortions.append(distortion)
    print("Number of Clusters:", k, "\tAverage Distortion:", distortion)

plt.plot(clusters, meandistortions, "bo-")  # Elbow Curve
plt.xlabel("K")
plt.ylabel("Distortion")
plt.title("Elbow method to select optimal K value")
